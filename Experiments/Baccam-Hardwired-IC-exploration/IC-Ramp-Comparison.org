* Background

The hardwired IC reparameterization ŷ(t) = z₀ + t · NN(t, θ) structurally
enforces the initial condition but interacts poorly with log transforms.
Near t = 0 the network must output ~(z(t) - z₀)/t, which diverges when the
true trajectory departs from z₀ linearly — a problem amplified when states
span many orders of magnitude (Baccam) but potentially benign when they
don't (SIR).

This notebook tests two hypotheses:

1. *The transform is the problem, not the reparameterization itself.*
   If hardwired IC works well on SIR (ZScoreTransform, states ~O(1)) but
   poorly on Baccam (LogTransform, states ~O(10⁻² to 10⁸)), the
   interaction with the log transform is the culprit.

2. *Replacing the linear ramp t with a saturating ramp σ(t) fixes the
   Baccam case.*  We compare three ramps:
   - =LuxMLPHardwiredIC=: σ(t) = t  (existing)
   - =LuxMLPTanhIC=: σ(t) = tanh(a·t)
   - =LuxMLPPowerIC=: σ(t) = (t/t_c)^α clamped to 1

The comparison matrix is:

| Model  | Transform | Soft IC | Hardwired (linear) | Tanh ramp | Power ramp |
|--------+-----------+---------+--------------------+-----------+------------|
| SIR    | ZScore    | ✓       | ✓                  | ✓         | ✓          |
| Baccam | Log₁₀     | ✓       | ✓                  | ✓         | ✓          |

#+begin_src julia-vterm :results nil

# ── Tanh ramp: σ(t) = tanh(a · t) ──────────────────────
# a controls the steepness. Default: reaches ~0.96 by t_span midpoint.

Base.@kwdef struct LuxMLPTanhIC <: AbstractArchitectureMode
    fd_step::Float32 = 1f-03
    a::Float32 = 1f0   # steepness; set per-problem below
end

function make_state_predictor_tanh_ic(mlp, st, y0_transformed::AbstractVector, a::Float32)
    function predict(ps, t_grid::AbstractMatrix)
        raw_nn = first(mlp(t_grid, ps.StateMLP, st))
        return y0_transformed .+ tanh.(a .* t_grid) .* raw_nn
    end
    return predict
end

function build_predictors(mode::LuxMLPTanhIC,
                          mlp, st, g_mlp, st_g,
                          y0_transformed)
    pred_state = make_state_predictor_tanh_ic(mlp, st, y0_transformed, mode.a)
    pred_deriv = make_deriv_predictor_fd(pred_state; h = mode.fd_step)
    pred_g     = make_g_predictor(g_mlp, st_g)
    return (predict_state = pred_state,
            predict_deriv = pred_deriv,
            predict_g     = pred_g,
            skip_ic_loss  = true)
end

# Reuse the same initialization as LuxMLP / LuxMLPHardwiredIC
function initialize_components(::LuxMLPTanhIC, rng, data, build_state_mlp_fn, build_g_mlp_fn)
    State_MLP = build_state_mlp_fn()
    ps_StateMLP, st_StateMLP = Lux.setup(rng, State_MLP)
    g_MLP = build_g_mlp_fn()
    ps_gMLP, st_gMLP = Lux.setup(rng, g_MLP)
    raw_ps = (StateMLP = ps_StateMLP, gMLP = ps_gMLP, ODE_par = data.ODE_par_init)
    return State_MLP, st_StateMLP, g_MLP, st_gMLP, raw_ps
end


# ── Power-law ramp: σ(t) = min((t / t_c)^α, 1) ────────
# α < 1 rises faster than linear near 0 → eliminates the 1/t singularity.
# t_c sets the timescale at which σ saturates to 1.

Base.@kwdef struct LuxMLPPowerIC <: AbstractArchitectureMode
    fd_step::Float32 = 1f-03
    α::Float32 = 0.5f0    # power-law exponent
    t_c::Float32 = 1f0     # saturation timescale; set per-problem below
end

function make_state_predictor_power_ic(mlp, st, y0_transformed::AbstractVector,
                                       α::Float32, t_c::Float32)
    function predict(ps, t_grid::AbstractMatrix)
        raw_nn = first(mlp(t_grid, ps.StateMLP, st))
        ramp = min.((max.(t_grid, 0f0) ./ t_c) .^ α, 1f0)
        # ramp = min.((t_grid ./ t_c) .^ α, 1f0)
        return y0_transformed .+ ramp .* raw_nn
    end
    return predict
end

function build_predictors(mode::LuxMLPPowerIC,
                          mlp, st, g_mlp, st_g,
                          y0_transformed)
    pred_state = make_state_predictor_power_ic(mlp, st, y0_transformed, mode.α, mode.t_c)
    pred_deriv = make_deriv_predictor_fd(pred_state; h = mode.fd_step)
    pred_g     = make_g_predictor(g_mlp, st_g)
    return (predict_state = pred_state,
            predict_deriv = pred_deriv,
            predict_g     = pred_g,
            skip_ic_loss  = true)
end

function initialize_components(::LuxMLPPowerIC, rng, data, build_state_mlp_fn, build_g_mlp_fn)
    State_MLP = build_state_mlp_fn()
    ps_StateMLP, st_StateMLP = Lux.setup(rng, State_MLP)
    g_MLP = build_g_mlp_fn()
    ps_gMLP, st_gMLP = Lux.setup(rng, g_MLP)
    raw_ps = (StateMLP = ps_StateMLP, gMLP = ps_gMLP, ODE_par = data.ODE_par_init)
    return State_MLP, st_StateMLP, g_MLP, st_gMLP, raw_ps
end

function SIR_model!(du, u, p, t)
    S, I, R = u
    du[1] = -p.β * S * I
    du[2] =  p.β * S * I - p.γ * I
    du[3] =  p.γ * I

#+begin_src julia-vterm :results nil

struct SIRParams{T<:AbstractFloat}
    β::T
    γ::T
end
SIRParams(nt::ComponentVector) = SIRParams(nt.β, nt.γ)

function SIRArchitectureRaw(y::AbstractMatrix{<:Real},
                            g::AbstractMatrix{<:Real},
                            p::SIRParams)
    S, I = y[1, :], y[2, :]
    g_S, g_R = g[1, :], g[2, :]

    dS = g_S
    dI = p.β .* S .* I .- p.γ .* I
    dR = g_R

    return permutedims(hcat(dS, dI, dR))
end

function Baccam_model!(du, u, p, t)
    T, I, V = u
    du[1] = -p.β * T * V

struct BaccamParams{T<:AbstractFloat}
    β::T
    δ::T
    c::T
end
BaccamParams(nt::ComponentVector) = BaccamParams(nt.β, nt.δ, nt.c)

function BaccamArchitectureRaw(y::AbstractMatrix{<:Real},
                               g::AbstractMatrix{<:Real},
                               p::BaccamParams)
    T, I, V = y[1,:], y[2,:], y[3,:]
    g_T, g_V = g[1,:], g[2,:]
    dT = g_T
    dI = p.β .* T .* V .- p.δ .* I
    dV = g_V .- p.c .* V
    return permutedims(hcat(dT, dI, dV))
end

The hardwired modes should all satisfy z(0) = z₀ exactly; the soft IC
mode satisfies it approximately.  Checking both models confirms the
structural guarantee holds regardless of transform.
