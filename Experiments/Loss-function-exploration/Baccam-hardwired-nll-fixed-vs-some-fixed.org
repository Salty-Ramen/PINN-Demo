* Goal

This notebook is fully self-contained and runnable.
It tests *Hardwired IC + Gaussian log-likelihood loss* for Baccam with two
ODE-parameter policies:

1. *All fixed:* keep =(β, δ, c)= fixed during training.
2. *Some fixed:* train β and c, keep δ fixed.

Template basis: =Experiments/Loss-function-exploration/Baccam-nll-vs-default-compare.org=,
but narrowed to only the Hardwired+NLL branch.

#+begin_src julia-vterm :results nil
function make_loss_tracker(architecture, ode_param_ctor, maxiters_s1;
                           log_every::Int = 100,
                           data_loss_fn = loss_Data)
    hist = (
        iter     = Int[],
        data_mse = Float32[],
        ode_mse  = Float32[],
        stage    = Int[],
    )

    function cb(state, l)
        state.iter % log_every != 0 && return false

        ps  = state.u
        ctx = state.p

        is_s2 = ctx isa PINNCtxStage2
        global_iter = is_s2 ? maxiters_s1 + state.iter : state.iter

        d_mse = data_loss_fn(ps, ctx)
        o_mse = is_s2 ? loss_ODE(ps, ctx, architecture, ode_param_ctor) : NaN32

        push!(hist.iter,     global_iter)
        push!(hist.data_mse, d_mse)
        push!(hist.ode_mse,  o_mse)
        push!(hist.stage,    is_s2 ? 2 : 1)

        return false
    end

    return cb, hist
end

** NLL loss functions

#+begin_src julia-vterm :results nil
const σ_NLL = 0.5f0

function loss_Data_NLL_transformed(ps, ctx)
    ẑ = ctx.predict_state(ps, ctx.t_train)
    return Statistics.mean(abs2, vec(ẑ - ctx.Y_train)) / (2f0 * σ_NLL^2)
end

function supervised_loss_NLL(ps, ctx, mode::AbstractHyperMode)
    hp = resolve_epsilons(ps, ctx, mode)
    penalty = hyper_penalty(ps, mode)

    L = loss_Data_NLL_transformed(ps, ctx) +
        l1_penalty_state(ps) / (hp.ϵ_L1_state^2 + 1f-6) +
        penalty

    if !ctx.skip_ic_loss
        L += loss_IC(ps, ctx) / (hp.ϵ_ic^2 + 1f-6)
    end

    return 0.5f0 * L
end

function unsupervised_loss_NLL(ps, ctx, mode::AbstractHyperMode,
                               architecture, ode_param_constructor)
    hp = resolve_epsilons(ps, ctx, mode)
    penalty = hyper_penalty(ps, mode)

    # ODE parameter box constraints are enforced at the optimizer level
    # (see build_box_bounds in Fitting.jl), not as a soft loss term here.
    L = loss_Data_NLL_transformed(ps, ctx) +
        l1_penalty_state(ps) / (hp.ϵ_L1_state^2 + 1f-6) +
        loss_ODE(ps, ctx, architecture, ode_param_constructor) /
            (hp.ϵ_ode^2 + 1f-6) +
        l1_penalty_g(ps) / (hp.ϵ_L1_g^2 + 1f-6) +
        penalty

    if !ctx.skip_ic_loss
        L += loss_IC(ps, ctx) / (hp.ϵ_ic^2 + 1f-6)
    end

    return 0.5f0 * L
end

#+begin_src julia-vterm
function Baccam_model!(du, u, p, t)
    T, I, V = u
    du[1] = -p.β * T * V
    du[2] =  p.β * T * V - p.δ * I
    du[3] =  p.p * I     - p.c * V
end

bac_true_params = (β = 2.7f-5, δ = 4.0f0, p = 1.2f-2, c = 3.0f0)
bac_u0     = Float32[4.0f8, 0.0f0, 9.3f-2]
bac_t_span = (0.0f0, 8.0f0)
bac_t_fine = collect(range(bac_t_span[1], bac_t_span[2], length = 2001))

bac_sol = solve(
    ODEProblem(Baccam_model!, bac_u0, bac_t_span, bac_true_params),
    Tsit5(); saveat = bac_t_fine
)
bac_sol_array = Array(bac_sol)

rng_bac = MersenneTwister(42)
bac_noise_σ = 0.5f0
bac_n_tp    = 10
bac_n_mice  = 12

bac_idx    = round.(Int, range(1, length(bac_t_fine), length = bac_n_tp))
bac_t_obs  = Float32.(bac_t_fine[bac_idx])
bac_Y_true = Float32.(bac_sol_array[:, bac_idx])

bac_ε = randn(rng_bac, Float32, 3, bac_n_tp, bac_n_mice)
bac_Y_noisy = clamp.(
    bac_Y_true .* exp.(bac_noise_σ .* bac_ε .- 0.5f0 * bac_noise_σ^2),
    0f0, Inf32
)
bac_Y_mean = dropdims(mean(bac_Y_noisy; dims = 3); dims = 3)

# Flatten full mouse dataset for NLL rows.
bac_Y_flat = Float32.(hcat([
    [bac_Y_noisy[s, k, m] for s in 1:3]
    for k in 1:bac_n_tp, m in 1:bac_n_mice
]...))
bac_t_flat = Float32.([
    bac_t_obs[k] for k in 1:bac_n_tp, m in 1:bac_n_mice
]) |> vec

println("Baccam training data: $(bac_n_tp) timepoints × $(bac_n_mice) mice")
println("NLL matrix shape: Y=$(size(bac_Y_flat)), t=$(length(bac_t_flat))")

#+begin_src julia-vterm :results nil
struct BaccamParams{T<:AbstractFloat}
    β::T; δ::T; c::T
end
BaccamParams(nt::ComponentVector) = BaccamParams(nt.β, nt.δ, nt.c)

function BaccamArchitectureRaw(y, g, p::BaccamParams)
    T, I, V = y[1,:], y[2,:], y[3,:]
    g_T, g_V = g[1,:], g[2,:]
    dT = g_T
    dI = p.β .* T .* V .- p.δ .* I
    dV = g_V .- p.c .* V
    return permutedims(hcat(dT, dI, dV))
end

bac_build_state_mlp() = Lux.Chain(
    Lux.Dense(1, 8, tanh_fast),
    Lux.Dense(8, 6, tanh_fast),
    Lux.Dense(6, 3),
)

bac_build_g_mlp() = Lux.Chain(
    Lux.Dense(1, 6, tanh_fast),
    Lux.Dense(6, 4, tanh_fast),

#+begin_src julia-vterm :results nil
function run_hardwired_nll_case(label::String, ode_policy;
                                 seed::Int,
                                 maxiters_stage1::Int = bac_maxiters_s1,
                                 maxiters_stage2::Int = bac_maxiters_s2)

    tracker_cb, hist = make_loss_tracker(
        BaccamArchitectureRaw, BaccamParams, maxiters_stage1;
        log_every = 100,
        data_loss_fn = loss_Data_NLL_transformed,
    )

    stats = @timed res = train(
        FixedHyper(),
        bac_data_nll_hard,
        BaccamArchitectureRaw,
        BaccamParams,
        bac_hp_hard_nll;
        transform                  = bac_transform,
        arch_mode                  = LuxMLPHardwiredIC(),
        ode_param_policy           = ode_policy,
        seed                       = seed,
        maxiters_stage1            = maxiters_stage1,
        maxiters_stage2            = maxiters_stage2,
        Opt_alg_stage1             = OptimizationOptimisers.Adam(bac_lr),
        Opt_alg_stage2             = OptimizationOptimJL.Fminbox(
                                         OptimizationOptimJL.LBFGS(m = 25)),
        build_state_mlp            = bac_build_state_mlp,
        build_g_mlp                = bac_build_g_mlp,
        callback_function          = tracker_cb,
        default_supervised_loss    = supervised_loss_NLL,
        default_unsupervised_loss  = unsupervised_loss_NLL,
    )

    return (label = label, res = res, hist = hist, time_s = stats.time)
end

run_specs = (
    hard_nll_all_fixed  = (policy = policy_all_fixed,  seed = 2101),
    hard_nll_some_fixed = (policy = policy_some_fixed, seed = 2102),
)

run_tasks = Dict{String, Task}()
for (label_sym, spec) in pairs(run_specs)
    label = String(label_sym)
    run_tasks[label] = Threads.@spawn begin
        println("[Thread $(Threads.threadid())] starting $label")
        run_hardwired_nll_case(label, spec.policy; seed = spec.seed)
    end
end

runs = Dict{String, Any}()
for (label, t) in run_tasks
    runs[label] = fetch(t)
end

run_order = ["hard_nll_all_fixed", "hard_nll_some_fixed"]
println("Completed ", length(runs), " runs.")
for label in run_order
    r = runs[label]
    p = r.res.params.ODE_par
    @printf("%-22s wall=%6.1fs  data=%.4e  ode=%.4e  β=%9.4e  δ=%9.4e  c=%9.4e\n",
        r.label, r.time_s, r.res.metrics.data_mse, r.res.metrics.ode_mse,
        p.β, p.δ, p.c)
end

* Interpretation checklist

- If *some-fixed* lowers ODE residual with similar NLL, selective freedom helps.
- If both are close, Hardwired+NLL fit may be driven mostly by state NN capacity.
- If all-fixed is much worse, parameter rigidity is likely limiting physics fit.
